\begin{frame}[fragile]{\textcolor{themeHtml}{\faCode}\ \ Tag-uri container vs void}
\vspace{6pt}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
{\footnotesize\bfseries\color{themeHtml}Container \,\textendash\, au continut}\\[4pt]
\begin{lstlisting}[style=html, numbers=none]
<p>text</p>
<a href="...">link</a>
<button>click</button>
<h1>Titlu</h1>
<div>...</div>
<span>...</span>
\end{lstlisting}

\column{0.48\textwidth}
{\footnotesize\bfseries\color{themeHtml}Void \,\textendash\, nu au continut}\\[4pt]
\begin{lstlisting}[style=html, numbers=none]
<img src="..." alt="...">
<br>   <hr>
<input type="text">
<meta charset="UTF-8">
<link rel="stylesheet">
\end{lstlisting}
{\footnotesize\color{textDim}\itshape Un singur tag. Fara \code{</tag>}.}
\end{columns}

\vspace{10pt}
\begin{center}

\end{center}
\end{frame}

\begin{frame}[fragile]{\textcolor{themeCss}{\faPaintBrush}\ \ \texttt{transition} vs \texttt{transform} -- nu confunda!}
\vspace{4pt}
\begin{columns}[T,onlytextwidth]
\column{0.49\textwidth}
  {\bfseries\color{themeCss}\code{transform} \,\textendash\, CE se schimba}\\[4pt]
  {\footnotesize Muta / scaleaza / roteste. Singur, schimbarea e \textbf{instant}.}
\begin{lstlisting}[style=css, basicstyle=\ttfamily\scriptsize, numbers=none]
.card { transform: scale(1.05); }
\end{lstlisting}

\column{0.49\textwidth}
  {\bfseries\color{themeCss}\code{transition} \,\textendash\, CUM se schimba}\\[4pt]
  {\footnotesize Nu schimba nimic singur. Animeaza \textbf{lin} orice schimbare.}
\begin{lstlisting}[style=css, basicstyle=\ttfamily\scriptsize, numbers=none]

\end{lstlisting}

\vspace{6pt}
{\footnotesize\bfseries\color{themeCss}Impreuna:}
\begin{lstlisting}[style=css, basicstyle=\ttfamily\scriptsize, numbers=none]

  transition: transform 0.2s ease;
}
.card:hover { transform: scale(1.05); }
\end{lstlisting}

\vspace{4pt}
{\footnotesize\color{textDim}\itshape \code{transition} se declara pe regula de baza (NU pe \code{:hover}). Analogie: \code{transform} = ,,muta cutia''; \code{transition} = ,,orice schimbare, alunec-o lin''.}
\end{frame}

const observer = new IntersectionObserver((entries) => {

      observer.unobserve(entry.target);
