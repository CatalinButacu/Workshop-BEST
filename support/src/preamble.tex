    enhanced,

    colframe=partcolor,

    sharp corners,
    arc=0pt,

    left=14pt, right=14pt, top=8pt, bottom=8pt,

      \node[partcolor!40, font=\fontsize{40}{40}\selectfont, anchor=north west]

    },
  ]
    #1
  \end{tcolorbox}
}
